In this problem, we are given an array with n elements. We want to find the minimum cost to make all of the elements equal.

## Solution

[Unofficial Editorial](https://codeforces.com/blog/entry/83295)

[Another solution and some proofs](https://www.geeksforgeeks.org/make-array-elements-equal-minimum-cost/)

Here is the code implemenation for the easiest approach.

```cpp
#include <bits/stdc++.h>
 
using namespace std;

#define endl "\n" 
//Make sure to define this 
//in case you accidentally use endl instead of "\n" to make a new line
//https://usaco.guide/general/fast-io/#fast-output
 
//variables used for the current problem
int n,median; 
vector<int> p;
long long ans,cnt;
 
void solve() {
	cin >> n;
	p.resize(n);
	for (int &x : p){
		cin >> x;
	}
	sort(p.begin(),p.end());
	med=p[n/2];
	for (const int &x : p){
		ans+=abs(median-x); //Calculate the cost to modify the stick length
	}
	cout << ans << endl;
	return;
}

int main(){
	ios_base::sync_with_stdio(false);
	cin.tie(nullptr);
	solve();
	return 0;
}
``` 
