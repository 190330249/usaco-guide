## Implementation

**Time Complexity:** $\mathcal{O}(N)$

```cpp
#include <iostream>
#include <bits/stdc++.h>

#define int int64_t

using namespace std;

vector<int> a,v; //cow i wants to visit cow a[i] and gets v[i] points
vector<vector<int>> graph; //graph[i] stores the cows that want to go to farm i
vector<bool> visited; //done

void mark(int y){
    if (visited[y]) return;
	visited[y] = true;
	for (int c : graph[y]) mark(c);
}

//Applies Floyd's algorithm
int min_in_cycle(int curr){
    int y = a[curr];
    int z = a[y];
    while(y != z){
        y = a[y];
        z = a[a[z]];
    }
    int min_v = v[y];
    y = a[y];
    while(y != z){
        min_v = min(min_v, v[y]);
        y = a[y];
    }
    mark(y);
    return min_v;
}

int32_t main()
{
    int n, ans = 0;
    cin >> n;
    v.resize(n);
    a.resize(n);
    visited.resize(n);
    graph.resize(n);
    for(int i = 0; i < n; i++){
        cin >> a[i] >> v[i];
        a[i]--;
        graph[a[i]].push_back(i); //Since cow i wants to visit farm a[i]
        ans += v[i];
    }

    //From each cycle, you can visit all except 1 cow. We make this cow the one
    //with the lowest v_i. We subtract this from our total answer.
    for (int i = 0; i < n; i++){
        if (!visited[i]) ans -= min_in_cycle(i);
    }

    cout << ans << "\n";
    return 0;
}
```
