## Implementation

**Time Complexity:** $\mathcal{O}(N^2)$

```java
import java.util.*;
import java.io.*;

public class Threesum {
	public static void main(String[] args) throws IOException {
		Kattio io = new Kattio("threesum");

		int N = io.nextInt();
		int Q = io.nextInt();
		final int MAX_VALUE = 1000000;
		int[] a = new int[N];
		long[][] dp = new long[N + 1][N + 1];
		int[] cnt = new int[2 * MAX_VALUE + 1];

		for (int i = 0; i < N; i++) {
			a[i] = io.nextInt();
		}

		for (int i = 0; i < N - 1; i++) {
			for (int j = 0; j < N; j++) {
				cnt[a[j] + MAX_VALUE] = 0;
			}

			for (int j = i+1; j < N; j++) {
				int k = -a[i] - a[j];
				if(k >= -MAX_VALUE && k <= MAX_VALUE) {
					dp[i + 1][j + 1] += cnt[k + MAX_VALUE];
				}

				cnt[a[j] + MAX_VALUE]++;
			}
		}
		
		for (int i = N; i >= 1; i--) {
			for (int j = i + 1; j <= N; j++) {
				dp[i][j] += dp[i + 1][j] + dp[i][j - 1] - dp[i + 1][j - 1];
			}
		}

		for (int i = 0; i < Q; i++) {
			int a1 = io.nextInt();
			int b = io.nextInt();

			io.println(dp[a1][b]);
		}

		io.close();
	}
	
	//CodeSnip{Kattio}
}
```
