## Explanation
Given the number of colored and uncolored nodes, the maximum number of edges that can exist is $c \cdot uc$ where $c$ represents the number of colored nodes, and $uc$ represents the number of uncolored nodes. 

This is because an edge can be drawn from each colored node to every other uncolored node. However, we already have $n - 1$ edges, so the answer is $(c \cdot uc) - (n - 1)$.

To find the number of colored and uncolored nodes, we can run a [DFS](https://usaco.guide/silver/dfs), incrementing the number of colored and uncolored nodes as we traverse the graph.

## Implementation

**Time Complexity:** $\mathcal{O}(N + M)$

```java
import java.util.*;
import java.io.*;

public class Bipartiteness {
	public static ArrayList<Integer> adj[];
	
	// counters for colored and uncolored nodes
	public static long[] cnt = {0, 0};

	public static void main(String[] args) throws IOException{
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		PrintWriter out = new PrintWriter(System.out);

		int n = Integer.parseInt(br.readLine());

		// +1 for 1-indexed nodes
		adj = new ArrayList[n + 1];
		
		for (int i = 0; i <= n; i++) adj[i] = new ArrayList<Integer>();

		StringTokenizer s;
		for (int i = 0; i < n - 1; i++) {
			s = new StringTokenizer(br.readLine());
			int u = Integer.parseInt(s.nextToken());
			int v = Integer.parseInt(s.nextToken());
			
			adj[u].add(v);
			adj[v].add(u);
		}
		
		dfs(1, 0, false);
		
		long ans = (cnt[0] * cnt[1]) - (n - 1); 
		
		out.println(ans);
		out.close();
	}

	public static void dfs(int node, int parent, boolean color) {
		cnt[color ? 1 : 0]++;

		for (int u : adj[node]) {
			// if the adjacent node isn't the parent,
			// visit that node with the opposite color
			if (u != parent) {
				dfs(u, node, !color);
			}
		}
	}
}
```

```cpp
#include <iostream>
#include <vector>
using namespace std;

const int MAX_N = 100000;

// +1 for 1-indexed nodes
vector<vector<int>> adj(MAX_N + 1);

// counters for colored and uncolored nodes
long long c = 0, uc = 0;

void dfs(int node, int parent, bool col) {
	if (col) c++;
	else uc++;

	for (int u : adj[node]) {
		// if the adjacent node isn't the parent,
		// visit that node with the opposite color
		if (u != parent) dfs(u, node, !col);
	}
}

int main() {
	int n;
	cin >> n;

	for (int i = 0; i < n - 1; i++) {
		int u, v;
		cin >> u >> v;
		adj[u].push_back(v);
		adj[v].push_back(u);
	}

	dfs(1, 0, 0);

	cout << (long long) (c * uc) - (n - 1) << "\n";
}
```
