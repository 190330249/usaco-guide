```py
import sys

sys.stdin = open("cowsignal.in","r")
sys.stdout = open("cowsignal.out","w")

M,N,K = map(int,input().split())
g = [input() for _ in range(M)]
for i in range(K*M):
	for j in range(K*N):
		print(g[i//K][j//K],end='')
	print()
```
