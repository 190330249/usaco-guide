## Solution

**Time Complexity:** $\mathcal{O}(N\cdot X)$

<JavaSection>

```java
import java.io.*;
import java.util.*;

public class bookShop {
	public static void main(String[] args) throws IOException {
		BufferedReader b = new BufferedReader(new InputStreamReader(System.in));
		StringTokenizer st = new StringTokenizer(b.readLine());
		int N = Integer.parseInt(st.nextToken());
		int X = Integer.parseInt(st.nextToken());

		// Read in the cost array.
		int[] costs = new int[N];
		st = new StringTokenizer(b.readLine());
		for (int i = 0; i < N; i++) {
			costs[i] = Integer.parseInt(st.nextToken());
		}

		// Read in the pages array.
		int[] pages = new int[N];
		st = new StringTokenizer(b.readLine());
		for (int i = 0; i < N; i++) {
			pages[i] = Integer.parseInt(st.nextToken());
		}

		// dp[i][j] is how many pages you have from the first i books,
		// after spending j money.
		int[][] dp = new int[N + 5][X + 5];
		for (int book = 1; book <= N; book++) {
			for (int money = 0; money <= X; money++) {
				dp[book][money] = dp[book - 1][money];
				int prev = money - costs[book - 1];
				if (prev >= 0) {
					//The total number of pages we have if we buy this book.
					int newState = dp[book - 1][prev] + pages[book - 1];
					dp[book][money] = Integer.max(dp[book][money], newState);
				}
			}
		}

		//Print the answer.
		System.out.println(dp[N][X]);
	}
}
```
