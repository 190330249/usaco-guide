## Implementation

**Time Complexity:** $\mathcal{O}(N)$

<CPPSection>

```cpp
#include <bits/stdc++.h>

#define MAXN 100001

using namespace std;

void setIO(string prob) {
	freopen((prob + ".in").c_str(), "r", stdin);
	freopen((prob + ".out").c_str(), "w", stdout);
}

int sigs[MAXN];
int presum[MAXN];

int main() {
	setIO("maxcross");
	int x, n, k, b, ret = INT_MAX;
	cin >> n >> k >> b;
	
	//input broken signals
	for (int i = 0; i < b; i++) {
		cin >> x; sigs[x]++;
	}
	
	// prefix sums precomputation
	for (int i = 0; i < n; i++) {
		presum[i + 1] = sigs[i + 1] + presum[i];
	}

	// find the answer
	for (int i = 0; i <= n - k; i++) {
		ret = min(ret, presum[i + k] - presum[i]);
	}
	cout << ret << '\n';
	return 0;
}
```

```java
import java.util.*;
import java.io.*;

public class maxcross {

	public static void main(String[] args) throws FileNotFoundException, IOException {
		BufferedReader in = new BufferedReader(new FileReader("maxcross.in"));
		PrintWriter pw = new PrintWriter(new BufferedWriter(new FileWriter("maxcross.out")));
		StringTokenizer st = new StringTokenizer(in.readLine());
		
		int n = Integer.parseInt(st.nextToken());
		int k = Integer.parseInt(st.nextToken());
		int b = Integer.parseInt(st.nextToken());
		int ret = n + 1;
		int[] sigs = new int[n + 1];
		int[] presum = new int[n + 1];

		// input broken signals
		for (int i = 0; i < b; i++) {
			st = new StringTokenizer(in.readLine());
			int x = Integer.parseInt(st.nextToken());
			sigs[x]++;
		}

		// prefix sums precomputation
		for (int i = 0; i < n; i++) {
			presum[i + 1] = sigs[i + 1] + presum[i];
		}

		// find the answer
		for (int i = 0; i <= n - k; i++) {
			ret = Math.min(ret, presum[i + k] - presum[i]);
		}
		pw.println(ret);
		pw.close();
	}
}
```
