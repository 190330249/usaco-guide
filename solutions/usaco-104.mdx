## C++ Implementation

```cpp
#include <bits/stdc++.h>

using namespace std;

int main () {
	int n, k;
	cin >> n >> k;
	vector<int> diff(n + 1);
	for (int i = 0; i < k; i++) { // read input and build difference array
		int l, r;
		cin >> l >> r;
		l--; // make zero-indexed
		r--;
		diff[l]++;
		diff[r + 1]--;
	}

	int sol[n];
	int tot = 0;
	for (int i = 0; i < n; i++) { // build prefix sum array
		tot += diff[i];
		sol[i] = tot;
	}
	sort (sol, sol + n); // sort to get median
	cout << sol[n/2] << '\n'; // output the median
}
```

## Java Implementation

```java
import java.io.*;
import java.util.*;

class haybaleStacking {
    public static void main (String[] args) throws IOException{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer input = new StringTokenizer(br.readLine());
        int n = Integer.parseInt(input.nextToken());
        int k = Integer.parseInt(input.nextToken());
        int[] differenceArray = new int[n + 1]; 

        for (int x = 0; x < k; x++) { // read input and build differenceArray
            StringTokenizer line = new StringTokenizer(br.readLine());
            int a = Integer.parseInt(line.nextToken());
            int b = Integer.parseInt(line.nextToken());
            a--; // make zero-indexed
            b--;
            differenceArray[a] = differenceArray[a] + 1;
            differenceArray[b + 1] = differenceArray[b + 1] - 1;
        }

        int[] prefixSum = new int[n + 1];
        int total = 0;
        for (int x = 0; x < n; x++) { // build prefixSum array
            total += differenceArray[x];
            prefixSum[x] = total;
        }
        Arrays.sort(prefixSum); // sort to get median
        System.out.println(prefixSum[prefixSum.length / 2]); // output the median
    }
}
```
