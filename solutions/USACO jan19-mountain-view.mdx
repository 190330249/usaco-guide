## Sorting Struct Using Comparator Function

```cpp
#include <bits/stdc++.h>

using namespace std;

#define FastIO ios_base::sync_with_stdio(0); cin.tie(0)
#define pb push_back
#define all(x) begin(x), end(x)
#define sor(x) sort(all(x))
#define ins insert
#define f first
#define s second
#define sz(x) (int)(x).size()
#define len(x) (int)(x).length()
#define FOR(i, a, b) for(int i=a; i<b; ++i)
#define ROF(i, b, a) for(int i=(b)-1; i>=a; --i)

void setIO(string fileName = ""){
	FastIO;
	if(len(fileName)){
		freopen((fileName+".in").c_str(), "r", stdin);
		freopen((fileName+".out").c_str(), "w", stdout);
	}
}

struct Mountain {
    int l, r;
};

bool cmp(Mountain a, Mountain b){
    if(a.l == b.l){
        return a.r > b.r;
    }
    return a.l < b.l;
}

int main(){
    setIO("mountains");
    int n; cin >> n;
    vector<Mountain> mountains;
    FOR(i, 0, n){
        int x, y; cin >> x >> y;
        Mountain m; m.l = x-y; m.r = x+y;
        mountains.pb(m);
    }
    sort(all(mountains), cmp);
    int mxRight = -1, ans = 0;
    for(int i = 0; i < n; i++) {
        if(mountains[i].r > mxRight){
            ans++;
            mxRight = max(mountains[i].r, mxRight);
        }
    }
    cout << ans << endl;
}
```
