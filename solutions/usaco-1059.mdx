```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
	int nums[7];
	for(int i = 0; i < 7; i++) {
		cin >> nums[i];
	}
	sort(nums, nums + 7);
	int a = nums[0], b = nums[1];
	int c = nums[6] - a - b;
	cout << a << ' ' << b << ' ' << c << '\n';
	return 0;
}
```

import java.util.*;
import java.io.*;
public class Main{
	public static void main(String[] args){

		int nums[] = new int[7];
		for(int i = 0; i < 7; i++){
			nums[i] = sc.nextInt();
		}
		Arrays.sort(nums);
		int a = nums[0], b = nums[1];
		int c = nums[6] - a - b;
		System.out.println(a + " " + b + " " + c);	
	} 
}
