**Time Complexity:** $\mathcal{O}(N \cdot T^2)$

```java
import java.util.*;
import java.io.*;

public class CowntactTracing {
	static boolean[] cowEndsInfected = new boolean[101];
	static int n;
	static int[] cowX = new int[251];
	static int[] cowY = new int[251];

	// this function simulates handshakes over time to see if data agrees
	// with this choice of patient_zero and K
	static boolean consistentWithData(int patientZero, int k) {
		boolean[] infected = new boolean[101];
		int[] numHandshakes = new int[101];

		infected[patientZero] = true;

		for (int t = 0; t <= 250; t++) {
			int x = cowX[t];
			int y = cowY[t];

			if (x > 0) {
				if (infected[x]) {
					numHandshakes[x]++;
				}
				if (infected[y]) {
					numHandshakes[y]++;
				}
				if (numHandshakes[x] <= k && infected[x]) {
					infected[y] = true;
				}
				if (numHandshakes[y] <= k && infected[y]) {
					infected[x] = true;
				}
			}
		}

		for (int i = 1; i <= n; i++) {
			if (infected[i] != cowEndsInfected[i]) {
				return false;
			}
		}

		return true; // return true at the end
	}

	public static void main (String[] args) throws IOException {
		Kattio io = new Kattio("tracing");

		n = io.nextInt();
		int T = io.nextInt();
		String s = io.next();

		for (int x = 1; x <= n; x++) {
			cowEndsInfected[x] = (s.charAt(x - 1) == '1'); // marks infected cows as true
		}

		for (int i = 0; i < T; i++) { // read input and store into array
			int t = io.nextInt();
			int x = io.nextInt();
			int y = io.nextInt();

			cowX[t] = x;
			cowY[t] = y;
		}

		boolean[] possibleI = new boolean[101];	
		boolean[] possibleK = new boolean[252];

		for (int i = 1; i <= n; i++) { // loop through each pair
			for (int k = 0; k <= 251; k++) {
				if (consistentWithData(i, k)) {
					possibleI[i] = true;
					possibleK[k] = true;
				}
			}
		}

		int lowerK = 251;
		int upperK = 0;
		int numPatientZero = 0;

		for (int k = 0; k <= 251; k++) {
			if (possibleK[k]) {
				upperK = k;
			}
		}

		for (int k = 251; k >= 0; k--) { 
			if (possibleK[k]) {
				lowerK = k;
			}
		}

		for (int i = 0; i <= n; i++) { // count number of patient zeros
			if (possibleI[i]) {
				numPatientZero++;
			}
		}

		io.print(numPatientZero + " " + lowerK + " "); // possible patient zeros, minimum of K
		if (upperK == 251) {
			io.println("Infinity"); // infinity if K = T is possible
		}
		else {
			io.println(upperK);
		}

		io.close();
	}

	//CodeSnip{Kattio}
}
```
