No need to prime factorize!

```cpp
int M,N;
vi todo[1000005];
bool done[1000005];

void dfs(int x) {
	if (done[x]) return;
	if (x == N*M) {
		ps("yes");
		exit(0);
	}
	done[x] = 1;
	trav(t,todo[x]) dfs(t);
}

int main() {
	setIO(); re(M,N);
	FOR(i,1,M+1) FOR(j,1,N+1) {
		int x; re(x);
		todo[i*j].pb(x);
	}
	dfs(1);
	ps("no");
}
```
