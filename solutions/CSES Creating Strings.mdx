The problem asks us to generate all permutations of a string in lexicographical order (or alphabetical order).

# Generate Permutations

Just generate all the permutations. If you want more details, check out [Complete Search with Recursion](/bronze/complete-rec). 

One small detail is that you need to sort the string before calling `next_permutation()` because the method generates strings in lexicographical order. If the string isn't sorted, then strings which are lexicographically better than the initial string won't be generated.

#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using vi = vector<int>;
#define pb push_back
#define rsz resize
#define all(x) begin(x), end(x)
#define sz(x) (int)(x).size()
using pi = pair<int,int>;
#define f first
#define s second
#define mp make_pair
void setIO(string name = "") { // name is nonempty for USACO file I/O
    ios_base::sync_with_stdio(0); cin.tie(0); // see Fast Input & Output
    if(sz(name)){
        freopen((name+".in").c_str(), "r", stdin); // see Input & Output
        freopen((name+".out").c_str(), "w", stdout);
    }
}

int main() {
    setIO();
    string s; cin >> s;    
    vector<string> out;
    sort(all(s));
    do {
        out.pb(s);
    } while(next_permutation(all(s)));
    cout << sz(out) << "\n";
    for(int i = 0; i < sz(out); i++){
        cout << out[i] << "\n";
    } 
}
