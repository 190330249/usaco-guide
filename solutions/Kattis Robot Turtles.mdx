## Explanation

We can use Dijkstra's, keeping track of the current length of the program
(`dist`), the direction the turtle faces (`dir`) and the moves it has already
made at each location (`moves`). If there is an ice castle in front of the
turtle, it should only be blasted if the turtle will move there immediately. If
the turtle doesn't move onto the square immediately, then it will either:

- return to the square later, in which case the turtle has wasted moves before
  returning
- never move onto the square, in which case the turtle wasted a move blasting
  the ice castle. 
So, we can only blast ice castles if the turtle moves forward immediately after.

There are 3 moves that a turtle can make at each location. It can move forward
if there is an empty square, or blast an ice castle and then move forward if
there is an ice castle. Or, the turtle can turn left or turn right. As we make
moves, we add the instructions to the string `moves`.

Once we reach the diamond, we can print out the `moves`.

```cpp
#include <iostream>
#include <string>
#include <queue>
#include <vector>
using namespace std;

struct State {
	// directions: 0 is up, 1 is right, 2 is down, 3 is left
	int dist, i, j, dir;
	string moves;
	bool operator>(const State& a) const
	{
		return dist > a.dist;
	}
};
bool ingrid(int a, int b)
{
	return a >= 0 && a < 8 && b >= 0 && b < 8;
}
int di[] = {-1, 0, 1, 0}, dj[] = {0, 1, 0, -1};

int main()
{
	string grid[8];
	for (int i = 0; i < 8; i++)
	{
		cin >> grid[i];
	}
	int dist[4][8][8]{}; // dir, i, j
	for (int i = 0; i < 4; i++)
	{
		for (int j = 0; j < 8; j++)
		{
			for (int k = 0; k < 8; k++)
			{
				dist[i][j][k] = 1e9;
			}
		}
	}
	dist[1][7][0] = 0;
	priority_queue<State, vector<State>, greater<State>> pq;
	pq.push({0, 7, 0, 1, ""});

	// use dijkstras to find the shortest path to the diamond
	while (!pq.empty())
	{
		State cur = pq.top();
		pq.pop();
		if (cur.dist != dist[cur.dir][cur.i][cur.j])
		{
			continue;
		}
		if (grid[cur.i][cur.j] == 'D')
		{
			cout << cur.moves << endl;
			return 0;
		}
		// move forward
		int newi = cur.i + di[cur.dir], newj = cur.j + dj[cur.dir];
		if (ingrid(newi, newj)
				&& (grid[newi][newj] == '.' || grid[newi][newj] == 'D')
				&& dist[cur.dir][cur.i][cur.j] + 1 < dist[cur.dir][newi][newj])
		{
			dist[cur.dir][newi][newj] = dist[cur.dir][cur.i][cur.j] + 1;
			pq.push({dist[cur.dir][newi][newj],
						newi, newj, cur.dir, cur.moves + "F"});
		}
		// blast ice castle and move forward
		if (ingrid(newi, newj)
				&& grid[newi][newj] == 'I'
				&& dist[cur.dir][cur.i][cur.j] + 2 < dist[cur.dir][newi][newj])
		{
			dist[cur.dir][newi][newj] = dist[cur.dir][cur.i][cur.j] + 2;
			pq.push({dist[cur.dir][newi][newj],
						newi, newj, cur.dir, cur.moves + "XF"});
		}
		// turn left or right
		for (int turn : {-1, 1})
		{
			int newdir = (cur.dir + turn + 4) % 4;
			string newmove = (turn == -1 ? "L" : "R");
			if (dist[cur.dir][cur.i][cur.j] + 1 < dist[newdir][cur.i][cur.j])
			{
				dist[newdir][cur.i][cur.j] = dist[cur.dir][cur.i][cur.j] + 1;
				pq.push({dist[newdir][cur.i][cur.j],
							cur.i, cur.j, newdir, cur.moves + newmove});
			}
		}
	}
	cout << "no solution" << endl;
	return 0;
}
```
