## C++ Implementation

```cpp
#include <iostream>
#include <cstdio>
#include <algorithm>
#include <vector>

using namespace std;

int n,k,ans,cnt; vector<int> a;

void solve(){
	cin >> n >> k;
	a.resize(n);
	for (int i=0;i<n;++i){
		cin >> a[i];
	}
	sort(a.begin(),a.end());
	for (int i=0;i<n;++i){
		cnt=1;
		for (int j=i+1;j<n;++j){
			if (a[j]-a[i]<=k) ++cnt;
			else break;
		}
		ans=max(cnt,ans);
	}
	cout << ans;
	return;
}

int main(){
	cin.tie(0)->sync_with_stdio(0);
	freopen("diamond.in","r",stdin);
	freopen("diamond.out","w",stdout);
	solve();
	return 0;
}
```
