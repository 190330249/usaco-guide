**Time Complexity:** $\mathcal{O}(H \cdot G)$

```cpp
#include <iostream>
#include <cstdio>
#include <vector>
using namespace std;

int energy(pair<int, int> a, pair<int, int> b)
{
	return (a.first - b.first) * (a.first - b.first) + (a.second - b.second) * (a.second - b.second);
}

int main() 
{
	freopen("checklist.in", "r", stdin);
	freopen("checklist.out", "w", stdout);
	int h, g;
	cin >> h >> g;
	vector<pair<int, int>> hcows(h), gcows(g);
	for (int i = 0; i < h; i++)
	{
		cin >> hcows[i].first >> hcows[i].second;
	}
	for (int i = 0; i < g; i++)
	{
		cin >> gcows[i].first >> gcows[i].second;
	}
	//dp[0][i][j] is min cost to visit i holsteins and j guernseys ending with holstein
	//dp[1][i][j] is min cost to visit i holsteins and j guernseys ending with guernsey
	long long dp[2][h + 1][g + 1];
	for (int i = 0; i <= h; i++)
	{
		for (int j = 0; j <= g; j++)
		{
			dp[0][i][j] = 1e18;
			dp[1][i][j] = 1e18;
		}
	}
	dp[0][1][0] = 0;
	for (int i = 0; i <= h; i++)
	{
		for (int j = 0; j <= g; j++)
		{
			if (i > 0 && j > 0)
			{
				//go from gcows[j - 1] to hcows[i - 1]
				dp[0][i][j] = min(dp[0][i][j],
								  dp[1][i - 1][j] + energy(hcows[i - 1], gcows[j - 1]));
				//go from hcows[i - 1] to gcows[j - 1]
				dp[1][i][j] = min(dp[1][i][j],
								  dp[0][i][j - 1] + energy(gcows[j - 1], hcows[i - 1]));
			}
			if (i > 1)
			{
				//go from hcows[i - 2] to hcows[i - 1]
				dp[0][i][j] = min(dp[0][i][j],
								  dp[0][i - 1][j] + energy(hcows[i - 1], hcows[i - 2]));
			}
			if (j > 1)
			{
				//go from gcows[j - 2] to gcows[j - 1]
				dp[1][i][j] = min(dp[1][i][j],
								  dp[1][i][j - 1] + energy(gcows[j - 1], gcows[j - 2]));
			}
		}
	}
	cout << dp[0][h][g] << endl;
	return 0;
}
```
