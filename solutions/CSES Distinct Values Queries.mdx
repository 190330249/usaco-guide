<Spoiler title = "Hint">
Sort the queries in a specific order before answering them.
</Spoiler>

<Spoiler title =  "Solution 1">

**Time Complexity:** $\mathcal{O}(N\log{N})$


Answer queries in order of the right indices. Use a BIT to keep track of indeces that contain distinct values. For every value, update the latest index for it and answer all queries at the current index. This ensures later indices are included in the BIT to accurately answer queries.

```cpp
#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define pi pair<int, int>
#define pb push_back
#define f first
#define s second


const int MX = (int)2e5+5;
int ans[MX], x[MX];
vector<pi> query[MX];
map<int,int> fst;

int fwt[MX];
void inc(int ind, int x){
    while(ind > 0 && ind < MX){
        fwt[ind] += x;
        ind += (ind&-ind);
    }
}
int sum(int ind){
    int ret = 0;
    while(ind > 0 && ind < MX){
        ret += fwt[ind];
        ind -= (ind&-ind);
    }
    return ret;
}
int main() {
    int N, Q; cin >> N >> Q;
    for(int i = 1; i <= N; i++){
        cin >> x[i];
    }
    for(int i = 0; i < Q; i++){
        int a,b; cin >> a >> b;
        query[b].pb({a,i});
    }
    for(int i = 1; i <= N; i++){
        int z = x[i];
        if (fst.count(z)) inc(fst[z],-1);
        fst[z] = i; inc(i,1);
        for(pi t : query[i]) ans[t.s] = sum(i) - sum(t.f-1);
    }
    for(int i = 0; i < Q; i++) cout << ans[i] << endl;
    return 0;
}
```
