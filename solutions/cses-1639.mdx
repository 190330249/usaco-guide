# Implementation
**Time Complexity:** $\mathcal{O}(NM)$

<LanguageSection>

<JavaSection>

```java
import java.io.*;
import java.util.*;

public class editDistance {
	public static void main(String[] args) throws IOException {
		Kattio io = new Kattio();
		char[] arr1 = io.next().toCharArray();
		char[] arr2 = io.next().toCharArray();
		int N = arr1.length;
		int M = arr2.length;

		/*
		 * dp[i][j] is the number of moves to change the first i letters
		 * of the string into the first j letters of result.
		 */
		int[][] dp = new int[N + 1][M + 1];
		for (int i = 0; i < dp.length; i++) {
			Arrays.fill(dp[i], Integer.MAX_VALUE);
		}
		dp[0][0] = 0;
		for (int i = 0; i <= N; i++) {
			for (int j = 0; j <= M; j++) {
				if (i != 0) {
					// Delete letter i - 1 from the string.
					dp[i][j] = Integer.min(dp[i][j], dp[i - 1][j] + 1);
				}
				if (j != 0) {
					// Add letter j - 1 of the result to the string.
					dp[i][j] = Integer.min(dp[i][j], dp[i][j - 1] + 1);
				}

				// Make letter i - 1 equal to letter j - 1 of the result.
				if (i != 0 && j != 0) {
					int sum = dp[i - 1][j - 1];
					if ((arr1[i - 1] != arr2[j - 1])) {
						sum++;
					}
					dp[i][j] = Integer.min(dp[i][j], sum);
				}
			}
		}
	   io.println(dp[N][M]);
	   io.close();
	}
	// Codesnip{Kattio};

}
```
