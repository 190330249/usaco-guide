## C++ Implementation

```cpp
#include <bits/stdc++.h>

using namespace std;

int main(){
  freopen("div7.in", "r", stdin);
  freopen("div7.out", "w", stdout);
  int n;
  cin >> n;
  int cur=0, ans=0;
  vector<int> first(7, -1);
  first[0]=0;
  for(int i=1;i<=n;i++){
    int x;
    cin >> x;
    cur=(cur+x)%7;
    if(first[cur]==-1){
      first[cur]=i;
    }else{
      ans=max(ans, i-first[cur]);
    }
  }
  cout << ans << endl;
}
```
