import java.util.*;
import java.io.*;
class Main {
	public static void main(String[] args) throws IOException {
		Scanner sc = new Scanner(new File("herding.in"));
		PrintWriter pw = new PrintWriter(new File("herding.out")); 
		int a = sc.nextInt();
		int b = sc.nextInt();
		int c = sc.nextInt();
		int temp;
		if (a > b) {
			temp = a;
			a = b;
			b = temp;
		}
		if (b > c) {
			temp = b;
			b = c;
			c = temp;
		}
		if (a > b) {
			temp = a;
			a = b;
			b = temp;
		}
		if (c == a + 2) {
			pw.println(0); 
			pw.print(a)
		}
		else if (b == a + 2 || c == b + 2) {
			pw.println(1); 
		}
		else {
			pw.println(2); 
		}
		pw.println(Math.max(b - a, c - b) - 1);
		pw.close();
	} 
}
