import java.util.*;
import java.io.*;
class Main {
	public static void main(String[] args) throws IOException {
		Scanner sc = new Scanner(new File("herding.in"));
		PrintWriter pw = new PrintWriter(new File("herding.out"));
		int[] ans = new int[3];
		ans[0] = sc.nextInt();
		ans[1] = sc.nextInt();
		ans[2] = sc.nextInt();
		Arrays.sort(ans);

		if (ans[2] == ans[0] + 2) {
			pw.println(0); 
		}
		else if (ans[1] == ans[0] + 2 || ans[2] == ans[1] + 2) {
			pw.println(1); 
		}
		else {
			pw.println(2); 
		}
		pw.println(Math.max(ans[1] - ans[0], ans[2] - ans[1]) - 1);
		pw.close();
	}
}
