import java.util.*;
import java.io.*;
public class Main{
	public static void main(String[] args){
		Scanner sc = new Scanner(System.in);
	  int[] array = new int[7];
		for(int i = 0; i<7; i++){
		  array[i] = sc.nextInt();  
		}
		int a = 0, b = 0, c = 0, abc = 0, bc, ba, ca = 0;
		int abcindex  = 0, aindex = 0, bcindex = 0;
		int smallest = 1000000000;
		int largest = 0;
 
		for(int i = 0; i<array.length; i++){
		  if(array[i]<=smallest){
			  a = array[i];
			  smallest = a;
			  aindex = i;
		  }
		  if(array[i]>=largest){
			  abc = array[i];
			  largest = abc;
			  abcindex = i;
		  }
		}
		bc = abc - a;
		array[aindex] = 0;
		array[abcindex] = 0;
		for(int i=0; i<array.length; i++){
		  if(array[i]==bc){
			bcindex = i;
		  }
		}
		array[bcindex] = 0;
		largest = 0;
		for(int i = 0; i<array.length; i++){
		  if(array[i]>=largest){
			ca = array[i];
			largest = ca;  
		  } 
		}
		c = ca-a;
		b = abc-a-c;
		System.out.println(a + " " + b + " " + c);
 
	  } 
}
