## C++ Implementation

```cpp
#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
using ll = long long;

int main(){
	ios_base::sync_with_stdio(0); cin.tie(0);
	freopen("square.in","r",stdin);
	freopen("square.out","w",stdout);
	ll x[4],y[4];
	for (int i=0;i<4;++i) {
		cin >> x[i] >> y[i];
	}
	ll a,b,c,d;
	a = min(x[0],x[2]);
	b = min(y[0],y[2]);
	c = max(x[1],x[3]);
	d = max(y[1],y[3]);
	ll ans = max((c-a)*(c-a),(d-b)*(d-b));
	cout << ans;
	return 0;
}
```
